def rag_summarize(query: str) -> str:
    # 向量检索得到相关文档片段（context）
    context = retrieve_from_vector_db(query)
    # 使用总结提示词让 LLM 生成简洁回答
    summary_prompt = f"""
你是一个专注于扫地/扫拖机器人知识总结的助手。
根据用户提问和参考资料，生成简洁准确的回答。

用户提问：{query}
参考资料：{context}

要求：
1. 回答必须完全基于参考资料，不编造。
2. 仅用中文，客观简洁。
3. 只输出回答内容本身，不要多余说明。
"""
    return call_llm(summary_prompt)
