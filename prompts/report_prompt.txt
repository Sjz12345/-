def fetch_external_data(user_id: str, month: str) -> str:
    # 从数据库查询原始报告数据
    raw_data = query_report_db(user_id, month)
    # 用提示词让 LLM 生成格式化报告
    report_prompt = f"""
你是一个专业的扫地/扫拖机器人报告写手。
根据以下使用记录数据，生成一份中文使用情况报告，并给出保养建议。

数据：{raw_data}

输出格式：
- 标题：黑马程序员扫地机器人使用情况报告与保养建议
- 内容：基于数据总结清洁效率、耗材状态、使用习惯，并给出具体保养建议。
- 不要直接输出原始数据。
"""
    return call_llm(report_prompt)
